#At first I looked at replacing standard pops by mirroring utopia.2551 . then I found out that you can add base = none to modify_species
namespace = gd_exotic_origin

### Shattered Ring Origin
country_event = {
	id = gd_exotic_origin.001
	hide_window = yes
	is_triggered_only = yes
	trigger = { has_origin = origin_shattered_ring }

	immediate = {
		if {
			limit = {
				NOR = {
					has_authority = auth_machine_intelligence
					has_trait = trait_pc_methane_preference
					has_trait = trait_pc_ammonia_preference
					has_trait = trait_pc_ash_preference
					has_trait = trait_pc_sulfur_preference
				}
			}
			every_owned_pop = {
				modify_species = {
					species = this
					ideal_planet_class = "pc_shattered_ring_habitable"
					base = none # makes this replace parent species
				}
			}
		}
	}
}
