#LL Methane
pc_ll_methane = {
	entity = "methane_planet"
	entity_scale = @planet_standard_scale
	icon = GFX_planet_type_methane
	#appearance
	picture = pc_methane
	atmosphere_color 		= hsv { 0.50 0.2 0.8 }	
	atmosphere_intensity 	= 1.0
	atmosphere_width 		= 0.5

	min_distance_from_sun = @habitable_min_distance
	max_distance_from_sun = @habitable_max_distance
	spawn_odds = 0
	city_color_lut = "gfx/portraits/misc/colorcorrection_methane.dds"
	modifier = {
		pop_environment_tolerance = 0.50
	}
	chance_of_ring = 0.2
	#planet defines
	climate = "exotic"
	colonizable = yes
	#colonization_tech = { }
	planet_size = { min = 16 max = 25 }
	moon_size = { min = 12 max = 16 }
	district_set = standard
	carry_cap_per_free_district = @carry_cap_normal
	starting_planet = no
	orbit_lines = yes
}
