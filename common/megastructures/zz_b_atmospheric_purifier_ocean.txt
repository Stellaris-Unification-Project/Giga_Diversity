	entity = "giga_consite_5_5_entity"
	construction_entity = "giga_consite_5_5_entity"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = no
	prerequisites = { giga_tech_terraform_toxic }

	build_time = 1260
	resources = {
		category = megastructures
		cost = {
			alloys = 2000
			influence = 75
		}
		upkeep = { energy = 5 }
	}

	potential = {
		has_technology = giga_tech_terraform_toxic
		has_ascension_perk = ap_world_shaper
		NOT = { has_global_flag = terraform_toxic_disabled }
		OR = {
			check_variable = { which = giga_terraform_toxic_cap value = 0 }						# No Cap
			check_variable = { which = giga_current_terraform_toxic value < giga_terraform_toxic_cap }
		}
	}

	possible = {
		custom_tooltip = { fail_text = "requires_inside_border"		is_inside_border = from }
		custom_tooltip = {
			fail_text = "requires_not_capped"
			from = {
				OR = {
					check_variable = { which = giga_terraform_toxic_cap value = 0 }						# No Cap
					check_variable = { which = giga_current_terraform_toxic value < giga_terraform_toxic_cap }
				}
			}
		}
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "must_build_around_toxic"
				OR = {
					is_planet_class = pc_toxic
					is_planet_class = pc_hothouse
					is_planet_class = pc_ethane_ocean
				}
			}
			custom_tooltip = { fail_text = "requires_no_anomaly"				has_anomaly = no }
			custom_tooltip = { fail_text = "requires_survey_not_habitable"		is_surveyed = { who = prev.from status = yes } is_planet_habitable = no }
			custom_tooltip = { fail_text = "requires_not_asteroid"				is_asteroid = no }
			custom_tooltip = { fail_text = "requires_not_star"					is_star = no }
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOT = { has_planet_flag = giga_terraforming_megastructure }
			}
		}
	}

	ai_weight = {
		weight = 300	# Default = 100
		#Does this benifit our main species? #Add check for aquatic trait once I know what the trait looks like.
		#modifier = { factor = 0 NOT = { owner_species = { has_trait = trait_pc_ash_preference } } }

		# Can we afford the upkeep

		# Do we want the production

		modifier = { factor = 3.00	fromfrom.planet = { planet_size >= 25 } }
		modifier = { factor = 2.00	fromfrom.planet = { planet_size < 25 planet_size >= 20 } }
		modifier = { factor = 1.50	fromfrom.planet = { planet_size < 20 planet_size >= 15 } }
		modifier = { factor = 0.00	fromfrom.planet = { planet_size < 15 } }

		# Do we lean towards this from federation/ethic/civic?
		modifier = {
			factor = 0
			from = {
				any_planet_within_border = {
					is_colonizable = yes
					planet_size > 15
				}
			}
		}

		# How many of this megastructure can the AI build?
		modifier = { factor = 0.00 from = { has_country_flag = has_recently_built_terraform_toxic } }
		modifier = {
			factor = 0
			from = {
				OR = {
					has_megastructure = terraform_toxic_0
					has_megastructure = terraform_toxic_1
					has_megastructure = terraform_toxic_2
					has_megastructure = terraform_toxic_3
					#GD additions
					has_megastructure = terraform_toxic_to_methane_0
					has_megastructure = terraform_toxic_to_methane_1
					has_megastructure = terraform_toxic_to_methane_2
					has_megastructure = terraform_toxic_to_methane_3
					has_megastructure = terraform_toxic_to_ash_0
					has_megastructure = terraform_toxic_to_ash_1
					has_megastructure = terraform_toxic_to_ash_2
					has_megastructure = terraform_toxic_to_ash_3
					has_megastructure = terraform_toxic_to_sulfur_0
					has_megastructure = terraform_toxic_to_sulfur_1
					has_megastructure = terraform_toxic_to_sulfur_2
					has_megastructure = terraform_toxic_to_sulfur_3
					has_megastructure = terraform_toxic_to_ammonia_0
					has_megastructure = terraform_toxic_to_ammonia_1
					has_megastructure = terraform_toxic_to_ammonia_2
					has_megastructure = terraform_toxic_to_ammonia_3
					has_megastructure = terraform_toxic_to_ocean_0
					has_megastructure = terraform_toxic_to_ocean_1
					has_megastructure = terraform_toxic_to_ocean_2
					has_megastructure = terraform_toxic_to_ocean_3
					has_megastructure = terraform_toxic_to_tidallylocked_0
					has_megastructure = terraform_toxic_to_tidallylocked_1
					has_megastructure = terraform_toxic_to_tidallylocked_2
					has_megastructure = terraform_toxic_to_tidallylocked_3

					has_megastructure = terraform_toxic_to_rogue_1
					has_megastructure = terraform_toxic_to_rogue_2
					has_megastructure = terraform_toxic_to_rogue_3
				}
			}
		}
	}

	on_build_complete = {
		save_event_target_as = giga_system
		fromfrom.planet = { save_event_target_as = giga_planet }
		from = {
			set_timed_country_flag = { flag = has_recently_built_terraform_toxic years = 20 }
			country_event = { id = giga_dialog.3901 }	# Notification
			change_variable = { which = giga_current_terraform_toxic value = 1 }
		}
		fromfrom.planet = {
			if = { limit = { exists = orbital_station } orbital_station = { dismantle = yes } }
			set_planet_flag = megastructure			# Vanilla uses to check if a megastrucure can be built here
			set_planet_flag = has_megastructure		# Vanilla uses to check if an orbital station can be built here
			set_planet_flag = giga_terraforming_megastructure
			#add_modifier = { modifier = "planetary_hab_imposs" days = -1}
			set_planet_flag = planetaryhabitatimpossible
		}
	}
}

terraform_toxic_to_ocean_3 = {
	entity = ""
	portrait = "GFX_megastructure_toxic_world"
	upgrade_from = { terraform_toxic_to_ocean_2 }

	build_time = 3600
	resources = {
		category = megastructures
		cost = { energy = 7500 }
	}

	potential = { NOT = { has_global_flag = terraform_toxic_disabled } }
	possible = { from = { has_technology = giga_tech_terraform_toxic } }

	ai_weight = { factor = 10 } # If we can build it, we will build it
	on_build_complete = {
		save_event_target_as = giga_system
		fromfrom.planet = { save_event_target_as = giga_planet }
		from = {
			country_event = { id = giga_dialog.3904 }	# Notification
			if = {
				limit = {
					NOR = {
						has_global_flag = giga_achievements_disabled
						has_country_flag = giga_achievement_58
					}
				}
				set_country_flag = giga_achievement_58
				set_timed_country_flag = { flag = giga_achievement_58_notification days = 30 }
				giga_achievement_sound = yes
			}
		}
		fromfrom.planet = {
			remove_planet_flag = giga_terraforming_megastructure
			remove_planet_flag = megastructure
			remove_planet_flag = has_megastructure
			change_pc = pc_ll_ocean
			add_modifier = { modifier = lifeless_ocean }
			giga_reroll_lifeless = yes
			set_planet_flag = used_to_be_toxic
			#TODO Create flag for used to be hot house
			#set_planet_flag = used_to_be_hothouse
			remove_planet_flag = planetaryhabitatimpossible
			remove_modifier = "planetary_hab_imposs"
			#remove autostripmining modifiers
			remove_modifier = giga_pd_ethane_planet
			#remove terraforming modifiers
			remove_modifier = terraforming_candidate
			remove_modifier = terraforming_ash
			remove_modifier = terraforming_ammonia
			remove_modifier = terraforming_sulfur
			remove_modifier = terraforming_methane
			#Remove special habitat modifiers
			remove_modifier = pdhabsubglacialocean
			remove_modifier = pdhabcavesystem
			remove_modifier = pdhabcalmwinds
			remove_modifier = pdhabvolcanicminerals
		}
		remove_megastructure = fromfrom
	}
}
