				fail_text = "must_build_around_barren_ship"

					is_planet_class = pc_barren
					is_planet_class = pc_barren_cold
					is_planet_class = pc_frozen
					is_planet_class = pc_subglacial
					is_planet_class = pc_diamond
					is_planet_class = pc_chthonian
					is_planet_class = pc_iron
					is_planet_class = pc_carbon
				}
			}
			custom_tooltip = {
				fail_text = "requires_no_existing_megastructure"
				NOT = { has_planet_flag = giga_terraforming_megastructure }
			}
		}
	}

		modifier = { factor = 3.00	fromfrom.planet = { planet_size >= 25 } }

		modifier = { factor = 1.50	fromfrom.planet = { planet_size < 20 planet_size >= 15 } }
		modifier = { factor = 0.00	fromfrom.planet = { planet_size < 15 } }

		# Do we lean towards this from federation/ethic/civic?
		modifier = {
			factor = 0
			from = {
				any_planet_within_border = {
					is_colonizable = yes
					planet_size > 15
				}
			}
		}

		# How many of this megastructure can the AI build?
		modifier = { factor = 0.00 from = { has_country_flag = has_recently_built_terraform_barren } }

					has_megastructure = terraform_barren_to_ash_1

					has_megastructure = terraform_barren_to_ammonia_3
					has_megastructure = terraform_barren_to_ammonia_3_restored
					has_megastructure = terraform_barren_to_ocean_0
					has_megastructure = terraform_barren_to_ocean_1
					has_megastructure = terraform_barren_to_ocean_2

			set_planet_flag = planetaryhabitatimpossible

		}
	}

terraform_barren_to_methane_3 = {
	entity = ""
	portrait = "GFX_megastructure_barren"
	upgrade_from = { terraform_barren_to_methane_2 }

	build_time = 3600
	resources = {
		category = megastructures
		cost = { energy = 7500 }
	}

	potential = { NOT = { has_global_flag = terraform_barren_disabled } }
	possible = { from = { has_technology = giga_tech_terraform_barren } }

	ai_weight = { factor = 10 } # If we can build it, we will build it
	on_build_complete = {
		save_event_target_as = giga_system
		fromfrom.planet = { save_event_target_as = giga_planet }
		from = {
			country_event = { id = giga_dialog.3804 }	# Notification
			if = {
				limit = {
					NOR = {
						has_global_flag = giga_achievements_disabled
						has_country_flag = giga_achievement_59
					}
				}
				set_country_flag = giga_achievement_59
				set_timed_country_flag = { flag = giga_achievement_59_notification days = 30 }
				giga_achievement_sound = yes
			}
		}
		fromfrom.planet = {
			remove_planet_flag = giga_terraforming_megastructure
			remove_planet_flag = megastructure
			remove_planet_flag = has_megastructure
			change_pc = pc_ll_methane
			add_modifier = { modifier = lifeless_methane }
			giga_reroll_lifeless = yes
			set_planet_flag = used_to_be_barren
			remove_modifier = "planetary_hab_imposs"
			remove_planet_flag = planetaryhabitatimpossible
			#remove special autostripmine modifiers
			remove_modifier = gd_pd_iron_planet
			remove_modifier = giga_pd_diamond_planet
			remove_modifier = giga_pd_ethane_planet
			remove_modifier = giga_pd_chthonian_planet
			#remove terraforming modifiers
			remove_modifier = terraforming_candidate
			remove_modifier = terraforming_ash
			remove_modifier = terraforming_ammonia
			remove_modifier = terraforming_sulfur
			remove_modifier = terraforming_methane
			#remove modifiers
			remove_modifier = pdhabsubglacialocean
			remove_modifier = pdhabcavesystem
			remove_modifier = pdhabcalmwinds
			remove_modifier = pdhabvolcanicminerals
		}
		remove_megastructure = fromfrom
	}
}
